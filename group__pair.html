template<typename pair >
using metal::first = typedef detail::first<pair>

#include <metal/pair/first.hpp>

Returns the first element of a Pair .

Usage

For any Value val

metal::first<val>;
Semantics:
If val is a Pair , then equivalent to
template<>
struct first<val> :
at<val, integer<0>>
{};
otherwise, equivalent to metal::nothing

Example

static_assert(std::is_same<metal::first_t<metal::pair<int, int*>>, int>::value, "");
static_assert(std::is_same<metal::first_t<metal::list<int, unsigned>>, int>::value, "");

Checks whether some Value is a Pair .

Usage

For any Value val

metal::is_pair<val>;
Semantics:
If val is a Pair , then equivalent to
template<>
struct is_pair<val> :
boolean<true>
{};
otherwise, equivalent to
template<>
struct is_pair<val> :
boolean<false>
{};

Example

static_assert(metal::is_pair_t<metal::pair<int, int*>>::value, "");
static_assert(metal::is_pair_t<metal::list<int, unsigned>>::value, "");
static_assert(!metal::is_pair_t<metal::list<int>>::value, "");
