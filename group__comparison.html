template<typename x , typename y >
using metal::equal_to = typedef detail::equal_to<x, y>

#include <metal/number/comparison/equal_to.hpp>

Checks whether two Numbers are equal.

Usage

For any Values val1 and val2

metal::equal_to<val1, val2>;
Semantics:
If both val1 and val2 are Numbers, then equivalent to
template<>
struct equal_to<val1, val2> :
boolean<val1::value == val2::value>
{};
otherwise, equivalent to metal::nothing

Example

static_assert(std::is_same<metal::equal_to_t<metal::integer<-10>, metal::number<long, 10>>, metal::boolean<false>>::value, "");
static_assert(std::is_same<metal::equal_to_t<metal::number<short, 0>, metal::boolean<false>>, metal::boolean<true>>::value, "");
