template<typename if_ , typename then , typename... else_>
using metal::conditional = typedef detail::conditional<if_, then, else_...>

#include <metal/optional/conditional.hpp>

Conditionally evaluates Optionals.

Usage

For any Optionals cond and opt1

metal::conditional<cond, opt1>;
Semantics:
Equivalent to
template<>
struct conditional<cond, opt1> :
conditional<cond, opt1, metal::nothing>
{};

For any Optionals cond, opt1 and opt2

metal::conditional<cond, opt1, opt2>;
Semantics:
If cond::type is a Number and cond::type::value is true, then equivalent to
template<>
struct conditional<cond, opt1, opt2> :
optional<opt1>
{};
otherwise, if cond::type is a Number and cond::type::value is false, then equivalent to
template<>
struct conditional<cond, opt1, opt2> :
optional<opt2>
{};
otherwise, equivalent to metal::nothing

For any Optionals opt1, opt2, opt3, opt4, ..., optn

metal::conditional<opt1, opt2, opt3, opt4, ..., optn>;
Semantics:
Equivalent to
template<>
struct conditional<opt1, opt2, opt3, opt4, ..., optn> :
conditional<opt1, opt2, conditional<opt3, opt4, ..., optn>>
{};

Example

template<typename num, typename min, typename max>
struct clip :
metal::conditional<
metal::greater<num, max>, max,
metal::less<num, min>, min,
num
>
{};
template<typename x, typename max, typename min>
using clip_t = typename clip<x, max, min>::type;
static_assert(std::is_same<
clip_t<metal::integer<12>, metal::integer<-10>, metal::integer<10>>,
metal::integer<10>
>::value, "");
static_assert(std::is_same<
clip_t<metal::integer<-12>, metal::integer<-10>, metal::integer<10>>,
metal::integer<-10>
>::value, "");
static_assert(std::is_same<
clip_t<metal::integer<0>, metal::integer<-10>, metal::integer<10>>,
metal::integer<0>
>::value, "");

#include <metal/optional/eval.hpp>

Evaluates an Optional .

A fallback Value may be specified in case the Optional is empty.

Usage

For any Optional opt

metal::eval<opt>;
Semantics:
Equivalent to
typename opt::type
Beware
Evaluating an empty optional leads to a compile time error.

For any Optional opt and Value val

metal::eval<opt, val>;
Semantics:
Equivalent to
conditional_t<is_just_t<opt>, opt, just<val>>;

Example

struct none
{};
struct some
{
struct type;
};
static_assert(std::is_same<metal::eval<some>, some::type>::value, "");
static_assert(std::is_same<metal::eval<some, void>, some::type>::value, "");
static_assert(std::is_same<metal::eval<none, void>, void>::value, "");

Checks whether an Optional represents some Value .

Usage

For any Optional opt

metal::is_just<opt>;
Semantics:
If opt::type is well defined and is a Value , then equivalent to

otherwise, equivalent to

Example

static_assert(!metal::is_just<none>::value, "");

template<typename opt >
using metal::optional = typedef detail::optional<opt>

#include <metal/optional/optional.hpp>

The standard representation of Optionals.

Usage

For any Optional opt

metal::optional<opt>;
Semantics:
If opt::type is well defined and is a Value , then equivalent to
template<>
struct optional<opt> :
just<typename opt::type>
{};
otherwise, equivalent to
template<>
struct optional<opt> :
nothing
{};
Tip
metal::optional<opt> is guaranteed to be inheritable.

Example

struct none
{};
struct some
{
struct type;
};
static_assert(!metal::is_just<metal::optional<none>>::value, "");
static_assert(metal::is_just<metal::optional<some>>::value, "");
static_assert(std::is_base_of<metal::nothing, metal::optional<none>>::value, "");
static_assert(std::is_base_of<metal::just<some::type>, metal::optional<some>>::value, "");
