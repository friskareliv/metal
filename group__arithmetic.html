template<typename... nums>
using metal::add = typedef detail::add<nums...>

#include <metal/number/arithmetic/add.hpp>

Computes the addition of Numbers.

Usage

For any Values val1, ..., valn

metal::add<val1, ..., valn>;
Semantics:
If all Values in [val1, ..., valn] are Numbers, then equivalent to
template<>
struct add<val1, ..., valn> :
number<
decltype(val1::value + ... + valn::value),
val1::value + ... + valn::value
>
{};
otherwise, equivalent to metal::nothing

Example

using x = metal::add_t<
metal::integer<-7>,
metal::number<long, 5>,
metal::number<short, 1>
>;
static_assert(std::is_same<x, metal::number<long, -1>>::value, "");

template<typename... nums>
using metal::pow = typedef detail::pow<nums...>

#include <metal/number/arithmetic/pow.hpp>

Computes the exponentiation of Numbers.

Usage

For any Values val1, val2, ..., valn

metal::pow<val1, val2, ..., valn>;
Semantics:
If all Values in [val1, ..., valn] are Numbers, and val1::value ** ... ** val{m-1}::value is nonzero for all nonpositive valm, then equivalent to
template<>
struct pow<val1, val2, ..., valn> :
number<
decltype(val1::value ** ... ** valn::value),
val1::value ** ... ** valn::value
>
{};
otherwise, equivalent to metal::nothing
Note
Borrowing from Fortran, x ** y means x raised to the power of y.
Beware
Due to inherent limitations of integer arithmetics, x ** y is always null for y < 0 and |x| > 1.

Example

using x = metal::pow_t<
metal::integer<-7>,
metal::number<long, 5>,
metal::number<short, 1>
>;
static_assert(std::is_same<x, metal::number<long, -16807>>::value, "");
static_assert(!metal::is_just<metal::pow<metal::integer<0>, metal::integer<-3>>>::value, "");
