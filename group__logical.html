template<typename... nums>
using metal::and_ = typedef detail::and_<nums...>

#include <metal/number/logical/and.hpp>

Computes the logical and of Numbers.

Usage

For any Values val1, ..., valn

metal::and_<val1, ..., valn>;
Semantics:
If all Values in [val1, ..., valn] are Numbers, then equivalent to
template<>
struct and_<val1, ..., valn> :
boolean<val1::value && ... && valn::value>
{};
otherwise, equivalent to metal::nothing

Example

using x = metal::and_t<
metal::integer<-7>,
metal::number<long, 5>,
metal::boolean<false>
>;
static_assert(std::is_same<x, metal::boolean<false>>::value, "");

#include <metal/number/logical/not.hpp>

Computes the logical negation of a Number .

Usage

For any Value val

metal::not_<val>;
Semantics:
If val is a Number , then equivalent to
template<>
struct not_<val> :
boolean<!val::value>
{};
otherwise, equivalent to metal::nothing

Example

static_assert(std::is_same<metal::not_t<metal::integer<-10>>, metal::boolean<false>>::value, "");
static_assert(std::is_same<metal::not_t<metal::number<short, 0>>, metal::boolean<true>>::value, "");
static_assert(std::is_same<metal::not_t<metal::boolean<false>>, metal::boolean<true>>::value, "");
static_assert(std::is_same<metal::not_t<metal::boolean<true>>, metal::boolean<false>>::value, "");
