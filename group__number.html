template<typename start , typename size = detail::nil, typename stride = detail::nil>
using metal::enumerate = typedef detail::enumerate<start, size, stride>

#include <metal/number/enumerate.hpp>

Generates a List of Numbers.

Usage

For any Value sz

metal::enumerate<sz>;
Semantics:
If sz is a Number , then equivalent to
template<>
struct enumerate<sz> :
enumerate<number<sz::value_type, 0>, sz>
{};
otherwise, equivalent to metal::nothing

For any Values st and sz

metal::enumerate<st, sz>;
Semantics:
Equivalent to
template<>
struct enumerate<st, sz> :
enumerate<st, sz, integer<1>>
{};

For any Values st, sz and sd

metal::enumerate<st, sz, sd>;
Semantics:
If st, sz and sd are Numbers and sz is equal to zero, then equivalent to
template<>
struct enumerate<st, sz, sd> :
list<>
{};
otherwise, if st, sz and sd are Numbers and sz is positive, then equivalent to
template<>
struct enumerate<st, sz, sd> :
list<
st,
number<st::value_type, st{} + sd{}>,
number<st::value_type, st{} + 2*sd{}>,
...,
number<st::value_type, st{} + (sz{} - 1)*sd{}>,
>
{};
otherwise, if st, sz and sd are Numbers and sz is negative, then equivalent to
template<>
struct enumerate<st, sz, sd> :
list<
st,
number<st::value_type, st{} - sd{}>,
number<st::value_type, st{} - 2*sd{}>,
...,
number<st::value_type, st{} - (1 - sz{})*sd{}>,
>
{};
otherwise, equivalent to metal::nothing

Example

using x = metal::enumerate_t<metal::integer<2>>;
using y = metal::enumerate_t<metal::integer<2>, metal::integer<3>>;
using z = metal::enumerate_t<metal::integer<2>, metal::integer<3>, metal::integer<-5>>;
static_assert(std::is_same<x, metal::list<metal::integer<0>, metal::integer<1>>>::value, "");
static_assert(std::is_same<y, metal::list<metal::integer<2>, metal::integer<3>, metal::integer<4>>>::value, "");
static_assert(std::is_same<z, metal::list<metal::integer<2>, metal::integer<-3>, metal::integer<-8>>>::value, "");

template<typename num >
using metal::is_number = typedef detail::is_number<num>

#include <metal/number/number.hpp>

Checks whether some Value is a Number .

Usage

For any Value val

metal::is_number<val>;
Semantics:
If val is a Number , then equivalent to
template<>
struct is_number<val> :
boolean<true>
{};
otherwise, equivalent to
template<>
struct is_number<val> :
boolean<false>
{};

Example

static_assert(metal::is_number_t<metal::number<short, 0>>::value, "");
static_assert(metal::is_number_t<metal::integer<-10>>::value, "");
static_assert(metal::is_number_t<metal::boolean<false>>::value, "");
static_assert(metal::is_number_t<metal::boolean<true>>::value, "");
static_assert(!metal::is_number_t<void>::value, "");
