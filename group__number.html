Computes the arithmetic addition of Numbers.

Usage

For any Numbers num_1, ..., num_n

using result = metal::add<num_1, ..., num_n>;
Returns
: Number
Semantics:
using result = metal::number<
decltype(num_1{} + ... + num_n{}),
num_1{} + ... + num_n{}
>;

Example

using num = metal::add<
metal::int_<-7>,
metal::number<short, 5>,
metal::number<long, 1>
>;
ASSERT(std::is_same<num, metal::number<long, -1>>);

template<typename from , typename to >
using metal::cast = typedef typename detail::_cast<from, to>::type

#include <metal/number/cast.hpp>

Performs non-narrowing integral conversion of Numbers.

Any Number may be cast to bool.

Usage

For any Number num and Value val

using result = metal::cast<num, val>;
Precondition
: val is an integral type
Returns
: Number of type val
Semantics:
using result = metal::number<val, val{num::value}>;

Example

using num = metal::int_<-10>;
ASSERT(std::is_same<metal::cast<num, bool>, metal::true_>);
ASSERT(std::is_same<metal::cast<num, short>, metal::number<short, -10>>);
ASSERT(!metal::is_invocable<metal::lambda<metal::cast>, num, unsigned>);

template<typename start , typename size , typename stride = int_<1>>
using metal::enumerate = typedef typename detail::_enumerate< start, cast<size, std::intmax_t>, cast<stride, std::intmax_t> >::type

#include <metal/number/enumerate.hpp>

Generates a sequence of Numbers.

Usage

For any Numbers st, sz and sd

using result = metal::enumerate<st, sz, sd>;
Returns
: List
Semantics:
If sz is positive, then
using result = metal::list<
st,
number<st::value_type, st{} + sd{}>,
number<st::value_type, st{} + 2*sd{}>,
...,
number<st::value_type, st{} + (sz{} - 1)*sd{}>,
>;
otherwise, if sz is negative, then
using result = metal::list<
st,
number<st::value_type, st{} - sd{}>,
number<st::value_type, st{} - 2*sd{}>,
...,
number<st::value_type, st{} - (1 - sz{})*sd{}>,
>;
otherwise
using result = metal::list<>;

Example

ASSERT(std::is_same<
metal::enumerate<metal::char_<'a'>, metal::int_<3>>,
metal::list<metal::char_<'a'>, metal::char_<'b'>, metal::char_<'c'>>
>);
ASSERT(std::is_same<
metal::enumerate<metal::int_<2>, metal::size_t<3>, metal::int_<-5>>,
metal::list<metal::int_<2>, metal::int_<-3>, metal::int_<-8>>
>);
ASSERT(std::is_same<
metal::enumerate<metal::size_t<2>, metal::int_<-3>, metal::int_<-5>>,
metal::list<metal::size_t<2>, metal::size_t<7>, metal::size_t<12>>
>);

A multi-clause conditional expression.

Usage

For any Numbers num_1, ..., num_n and Values val_1, ..., val_n+1

using result = metal::if<num_1, val_1, ..., num_n, val_n, val_n+1>;
Returns
: Value
Semantics:
If num_i{} == true and num_j{} == false for all j < i, then
using result = val_i;
otherwise, if num_i{} == false for all i in [1, ..., n], then
using result = val_n+1;

Example

metal::if_<metal::false_, int, metal::int_<0>, int*, void>,
void
>);
ASSERT(std::is_same<
metal::if_<metal::false_, int, metal::int_<1>, int*, void>,
int*
>);
ASSERT(std::is_same<
metal::if_<metal::true_, int, metal::int_<1>, int*, void>,
int
>);

template<typename head , typename... tail>
using metal::pow = typedef typename detail::_pow<head, tail...>::type

#include <metal/number/pow.hpp>

Computes the arithmetic exponentiation of Numbers.

Usage

For any Numbers num_1, ..., num_n

using result = metal::pow<num_1, ..., num_n>;
Precondition
: num_1{} ** ... ** num_m-1{} is nonzero for all negative num_m
Returns
: Number
Semantics:
using result = metal::number<
decltype(num_1{} ** ... ** num_n{}),
num_1{} ** ... ** num_n{}
>;
Borrowing from Fortran, x ** y should be understood as x raised to the power of y.
x ** y is always null for y < 0 and |x| > 1 due to inherent limitations of integer arithmetics.

Example

using num = metal::pow<
metal::int_<-7>,
metal::number<short, 5>,
metal::number<long, 1>
>;
ASSERT(std::is_same<num, metal::number<long, -16807>>);
ASSERT(!metal::is_invocable<metal::lambda<metal::pow>, metal::int_<0>, num>);
